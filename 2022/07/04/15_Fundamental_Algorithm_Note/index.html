数组高精度乘法

#include <stdio.h>
#include <string.h>

int main() {
    char numberN[1500], numberM[1500];
    scanf("%s%s", numberN, numberM);
    int n = strlen(numberN), m = strlen(numberM);
    int a[n], b[m];
    int i, j;
    for (i = 0, j = n - 1; i < n; i++, j--) {
        a[i] = numberN[j] - '0';
    }
    for (i = 0, j = m - 1; i < m; i++, j--) {
        b[i] = numberM[j] - '0';
    }//储存
    int c[3000] = {};
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++) {
            c[i + j] += a[i] * b[j];//核心 可以手动跟踪一下 虽然其实就是列竖式罢了
        }
    }
    for (i = 0; i < n + m; i++) {
        if (c[i] >= 10) {
            c[i + 1] += c[i] / 10;
            c[i] %= 10;//进位
        }
    }
    for (j = 2999; j > 0; j--) {
        if (c[j] != 0)
            break;//确定乘积的位数
    }
    for (i = j; i >= 0; i--) {
        printf("%d", c[i]);//逆序输出
    }
    return 0;
}

滑动窗口算法

@知乎回答：什么是「滑动窗口算法」？ 有给定窗口长度&不给定两种分支。

第一种一般是寻找满足条件的长度最小的连续子数组；第二种一般是寻找包含给定字符串的最短子串，均不适合进行计数。
